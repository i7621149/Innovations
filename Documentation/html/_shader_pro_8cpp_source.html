185 void ShaderPro::loadImage(int _textureUnit, TextureData _texture, GLenum _type)
186 {
187  std::cout << "texture ID: " << _texture.id <<std::endl;
188  std::cout << "texture unit: " << _textureUnit << std::endl;
189 
190  QImage image;
191  // load given texture files
192  bool loaded=image.load(_texture.textureSource.c_str());
193  if(loaded == true)
194  {
195  int width=image.width();
196  int height=image.height();
197 
198  unsigned char *data = new unsigned char[ width*height*3];
199  unsigned int index=0;
200  // reversed height so it loads as is used in shader
201  for( int y=height-1; y>=0; --y)
202  {
203  for( int x=0; x<width; ++x)
204  {
205  // getting RGB from the image
206  QRgb colour=image.pixel(x,y);
207  data[index++]=qRed(colour);
208  data[index++]=qGreen(colour);
209  data[index++]=qBlue(colour);
210  }
211  }
212 
213  // set active texture unit
214  glActiveTexture(GL_TEXTURE0 + _textureUnit);
215 
216  // is this the right place??
217  glBindTexture(_type, _texture.id);
218 
219  // load texture
220  glTexImage2D(_type,0,GL_RGB,width,height,0,GL_RGB,GL_UNSIGNED_BYTE,data);
221 
222  // setting up mipmap parameters
223  // not sure whether the first should be GL_LINEAR or GL_LINEAR_MIPMAP_LINEAR?
224  glTexParameteri(_type, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
225  glTexParameteri(_type, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
226 
227  // set wrapping parameters for textures
228  glTexParameteri(_type, GL_TEXTURE_WRAP_S, GL_REPEAT);
229  glTexParameteri(_type, GL_TEXTURE_WRAP_T, GL_REPEAT);
230  // creating mipmaps, needs to be done after texture is loaded
231  glGenerateMipmap(_type);
232 
233  // calculate texture channel name
234  // this allows for easier adding of textures, as they can all be called iChannelX where X is their number
235  std::ostringstream convertStream;
236  convertStream << _textureUnit;
237  std::string channelName = "iChannel" + convertStream.str();
238 
239  // get texture unit location and set uniform up
240  GLuint texLocation = glGetUniformLocation(m_progID, channelName.c_str());
241 
242  glUniform1i(texLocation, _textureUnit);
243 
244  if(texLocation == -1)
245  {
246  // warning, as it might not exist if it is never used in shader
247  std::cerr << "WARNING: cannot get uniform location of " << channelName << "\nPerhaps it is not being used in the shader?" << std::endl;
248  }
249  else
250  {
251  // print info to confirm texture loaded
252  std::cout << "Loaded texture to " << channelName.c_str() << std::endl;
253  }
254  // clean up
255  delete[] data;
